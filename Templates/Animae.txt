int n, m;

bool read () {
	if (scanf("%d%d", &n, &m) != 2)
		return false;
	return true;
}

vector<vector<int>> ans;

void transpose(){
	vector<vector<int>> tmp(sz(ans[0]), vector<int>(sz(ans)));
	forn(i, sz(ans)) forn(j, sz(ans[0])) tmp[j][i] = ans[i][j];
	ans = tmp;
}

void solve() {
	if (n == 1 && m == 1){
		puts("1");
		return;
	}
	if (min(n, m) == 1 && max(n, m) <= 3){
		puts("-1");
		return;
	}
	if (n == 2 && m == 2){
		puts("-1");
		return;
	}
	if (n == 2 && m == 3){
		puts("3 6 2");
		puts("5 1 4");
		return;
	}
	if (n == 3 && m == 2){
		puts("3 5");
		puts("6 1");
		puts("2 4");
		return;
	}
	if (n == 3 && m == 3){
		puts("3 6 2");
		puts("9 1 8");
		puts("5 7 4");
		return;
	}
	bool fl = m < n;
	if (m < n) swap(n, m);
	ans.clear();
	forn(i, n){
		ans.pb({});
		for (int j = (m + 1) / 2 - 1; j >= 0; --j) ans.back().pb(i * m + j * 2 + 1);
		for (int j = m - 1; j >= (m + 1) / 2; --j) ans.back().pb(i * m + (j - (m + 1) / 2) * 2 + 2);
	}
	if (fl) transpose();
	for (auto it : ans){
		for (auto jt : it)
			printf("%d ", jt);
		puts("");
	}
}
